### Install as a package

<RefSubLayout.EducationRow>
  <RefSubLayout.Details>

    Install `@supabase/ssr` via your package manager. It has `@supabase/supabase-js` as a peer dependency, so install both.

  </RefSubLayout.Details>

  <RefSubLayout.Examples>

    <Tabs
      size="small"
      type="underlined"
      defaultActiveId="npm"
      queryGroup="platform"
    >
      <TabPanel id="npm" label="npm">

        ```sh Terminal
        npm install @supabase/ssr @supabase/supabase-js
        ```

      </TabPanel>
      <TabPanel id="yarn" label="Yarn">

        ```sh Terminal
        yarn add @supabase/ssr @supabase/supabase-js
        ```

      </TabPanel>
      <TabPanel id="pnpm" label="pnpm">

        ```sh Terminal
        pnpm add @supabase/ssr @supabase/supabase-js
        ```

      </TabPanel>
      <TabPanel id="bun" label="Bun">

        ```sh Terminal
        bun add @supabase/ssr @supabase/supabase-js
        ```

      </TabPanel>
    </Tabs>

  </RefSubLayout.Examples>
</RefSubLayout.EducationRow>
